





import requests
import time

url = "http://quotes.toscrape.com"

request_times = []
is_bot_behavior = True

print("Sending requests to:", url)

start_total = time.time()

for i in range(10):
    start = time.time()

    requests.get(url)

    end = time.time()
    request_times.append(end - start)

    if is_bot_behavior:
        time.sleep(0)   # bot: no delay
    else:
        time.sleep(1)   # human: delay

end_total = time.time()

# Metrics
avg_request_time = sum(request_times) / len(request_times)
total_time = end_total - start_total

if is_bot_behavior:
    result = "Bot Detected"
else:
    result = "Human User Detected"

print("Average Request Time:", avg_request_time)
print("Total Time:", total_time)
print("Result:", result)
