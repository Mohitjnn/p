import pandas as pd

df = pd.read_csv("data.csv")

df = df.drop_duplicates()          # remove duplicates
df = df.dropna()                   # remove missing values
df["Age"] = pd.to_numeric(df["Age"], errors="coerce")
df = df.dropna()                   # remove invalid age rows
df["Name"] = df["Name"].str.lower()

print(df)
